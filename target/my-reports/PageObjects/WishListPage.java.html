	WebElement AddToCart;
	
	@FindBy (xpath="//div[@class=\"alert alert-success alert-dismissible\"]")
	WebElement GetMsg;
	
	@FindBy (xpath="(//i[@class=\"fas fa-shopping-cart\"])[1]")
